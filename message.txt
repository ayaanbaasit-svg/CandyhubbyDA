-- // [4] UTILITY FUNCTIONS //

local function getCharacter()
    return LocalPlayer.Character
end

local function getHumanoid()
    local char = getCharacter()
    if not char then return end
    return char:FindFirstChildOfClass("Humanoid")
end

local function getHRP()
    local char = getCharacter()
    if not char then return end
    return char:FindFirstChild("HumanoidRootPart")
        or char:FindFirstChild("UpperTorso")
end

local function getPing()
    local ok, val = pcall(function()
        return Stats.Network.ServerStatsItem["Data Ping"]:GetValue()
    end)
    return ok and math.floor(val) or 0
end

local function getSpeed()
    local hrp = getHRP()
    if hrp then
        local vel = hrp.Velocity
        return math.floor(math.sqrt(vel.X^2 + vel.Z^2) * 10) / 10
    end
    return 0.0
end

local function findMyPlot()
    local plotsFolder = workspace:FindFirstChild("Plots")
    if not plotsFolder then return nil end
    for _, plot in ipairs(plotsFolder:GetChildren()) do
        local sign = plot:FindFirstChild("PlotSign")
        if sign then
            local yb = sign:FindFirstChild("YourBase")
            if yb and yb.Enabled then
                return plot
            end
        end
    end
    return nil
end

local function getPlotPodiums(plot)
    if not plot then return {} end
    local ap = plot:FindFirstChild("AnimalPodiums")
    if not ap then return {} end
    return ap:GetChildren()
end

local function disableAnimations()
    local hum = getHumanoid()
    if not hum then return end
    local animator = hum:FindFirstChildOfClass("Animator")
    if animator then
        for _, track in ipairs(animator:GetPlayingAnimationTracks()) do
            track:Stop()
        end
    end
end

local function enableNoPlayerCollision(active)
    local char = getCharacter()
    if not char then return end
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CollisionGroup = active and "NoPlayerCollision" or "Default"
        end
    end
end

local function autoTeleportToAnimal()
    local myPlot = findMyPlot()
    if not myPlot then return end
    local hrp = getHRP()
    if not hrp then return end
    local podiums = getPlotPodiums(myPlot)
    local closest, dist = nil, CONFIG.AutoTPRadius
    for _, pod in ipairs(podiums) do
        local base = pod:FindFirstChild("Base")
        local spawn = base and base:FindFirstChild("Spawn")
        if spawn then
            local d = (hrp.Position - spawn.Position).Magnitude
            if d < dist then
                dist = d
                closest = spawn
            end
        end
    end
    if closest then
        local pp = closest:FindFirstChild("PromptAttachment")
        local prompt = pp and pp:FindFirstChildOfClass("ProximityPrompt")
        if prompt then
            prompt.Enabled = false
            prompt.HoldDuration = 0
            prompt.RequiresLineOfSight = false
            prompt.MaxActivationDistance = 100
            fireproximityprompt(prompt)
            task.delay(0.1, function()
                prompt.Enabled = true
            end)
        end
    end
end

-- // [6] FUNCTIONALITY //

local function setToggle(name, value)
    local t = Toggles[name]
    if not t then return end
    t.active = value
    t.stroke.Color = value and CONFIG.Colors.ButtonStroke or CONFIG.Colors.ButtonStrokeOff
    State[name] = value
end

local function bindToggle(name, callback)
    local t = Toggles[name]
    if not t then return end
    t.button.MouseButton1Click:Connect(function()
        local newVal = not t.active
        setToggle(name, newVal)
        if callback then callback(newVal) end
    end)
    -- Apply default
    setToggle(name, t.active)
    if callback then callback(t.active) end
end

local function applyWalkSpeed(speed)
    local hum = getHumanoid()
    if hum then hum.WalkSpeed = speed end
end

-- // [7] INITIALIZATION //

local function init()
    createGUI()
    setupFeatures()

    -- Load luarmor module
    pcall(function()
        loadstring(game:HttpGet(CONFIG.LuarmorURL, true))()
    end)

    -- Heartbeat loop
    HeartbeatConn = RunService.Heartbeat:Connect(function(dt)
        FPSCounter = FPSCounter + 1
        local now = tick()
        if now - LastFPSTime >= 1 then
            local fps = math.floor(FPSCounter / (now - LastFPSTime))
            local ping = getPing()
            if FPSLabel then
                FPSLabel.Text = "FPS: " .. fps .. "\nPing: " .. ping .. "ms"
            end
            FPSCounter = 0
            LastFPSTime = now
        end

        -- Overhead speed
        local speed = getSpeed()
        if OverheadSpeedLabel then
            OverheadSpeedLabel.Text = "SPEED: " .. string.format("%.1f", speed)
        end

        -- Auto Steal loop
        if State["AutoSteal"] then
            autoTeleportToAnimal()
        end

        -- Auto TP loop
        if State["AutoTP"] then
            autoTeleportToAnimal()
        end

        -- Walk speed maintenance
        if State["NormalSpeed"] then applyWalkSpeed(CONFIG.NormalWalkSpeed)
        elseif State["LoggerSpeed"] then applyWalkSpeed(CONFIG.LoggerWalkSpeed)
        elseif State["DesyncSpeed"] then applyWalkSpeed(CONFIG.DesyncWalkSpeed)
        end
    end)
end

init()
